#include<iostream>
using namespace std;
#define ll long long int

ll comb(ll n,ll r){
    return (n*(n-1))/2;
}

int main() {
    int t;
    cin>>t;
    for(int i=0;i<t;i++){
        ll n;
        cin>>n;
        ll arr[n],sum=0,freq[n]={0};
        for(ll j=0;j<n;j++){
            cin>>arr[j];
            sum = sum + arr[j];
            sum=sum%n;
            sum = (sum+n)%n;
            freq[sum]++;
        }
        ll ans = 0;
        freq[0]++;
        for(ll j=0;j<n;j++){
            if(freq[j]>1)
            ans = ans + comb(freq[j],2);
        }
        cout<<ans<<endl;
    }
	return 0;
}
