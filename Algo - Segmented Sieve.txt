#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define mod 1000000007
int main()
 {
	int t;
	cin>>t;
	int size = 100000;
	bool arr[size+1];
	for(int i=0;i<=size;i++){
	    if(i%2)
	    arr[i] = true;
	    else
	    arr[i] = false;
	}
	arr[1] = false;
	arr[2] = true;
	for(int i=3;i<=sqrt(size);i++){
	    if(arr[i]){
	        for(int j=i*i;j<=size;j+= 2*i)
	        arr[j] = false;
	    }
	}
	vector<int>prime;
	prime.push_back(2);
	for(int i=3;i<=size;i+=2){
	    if(arr[i])
	    prime.push_back(i);
	}
	for(int i=0;i<t;i++)
	{
	    ll l,r;
	    cin>>l>>r;
	    bool segment[r-l+1];
	    for(int j=0;j<r-l+1;j++)
	    segment[j] = true;
	    for(int j=0;j<prime.size();j++){
	        for(int k=(l/prime[j])*prime[j];k<=r;k+=prime[j]){
	            if(k >= l and sqrt(k) >= prime[j])
	            segment[k-l] = false;
	        }
	    }
	    ll sum = 1;
	    for(int j=0;j<r-l+1;j++){
	        if(segment[j])
	        sum = ((sum%mod) * ((j+l)%mod))%mod;
	    }
	    cout<<sum<<endl;
	}
	
	return 0;
}
