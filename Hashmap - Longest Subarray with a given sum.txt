#include<bits/stdc++.h>
using namespace std;

int main(){
    int n,k;
    cin>>n>>k;
    int arr[n];
    for(int i=0;i<n;i++)
    cin>>arr[i];
    int pre = 0;
    unordered_map<int,int>m;
    m[pre] = -1;
    for(int i=0;i<n;i++){
        pre = pre + arr[i];
        arr[i] = pre;
    }
    int length = 0;
    for(int i=0;i<n;i++){
        pre = arr[i] - k;
        m[pre] = i;
    }
    for(int i=0;i<n;i++){
        if(m.count(arr[i])){
            length = max(length,abs(i-m[arr[i]]));
        }
    }
    cout<<length<<endl;
    return 0;
}
