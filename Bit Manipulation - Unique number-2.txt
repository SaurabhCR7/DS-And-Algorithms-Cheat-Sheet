#include<iostream>
using namespace std;
int main() {
    int n;
    cin>>n;
    int arr[n];
    int xort = 0,pos=0;
    for(int i=0;i<n;i++){
        cin>>arr[i];
        xort = xort^arr[i];
    }
    int temp = xort;
    while(temp&1 != 1){
        pos++;
        temp = temp>>1;
    }
    int num,b=0;
    for(int i=0;i<n;i++){
        num = 1<<pos;
        if(arr[i]&num > 0)
        b = b^arr[i];
    }
    int a = xort^b;
    cout<<min(a,b)<<" "<<max(a,b)<<endl;
	return 0;
}
