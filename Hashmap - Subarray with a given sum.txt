#include<bits/stdc++.h>
using namespace std;
int main()
 {
	int t;
	cin>>t;
	for(int i=0;i<t;i++)
	{
	    int n;
	    cin>>n;
	    int arr[n];
	    for(int j=0;j<n;j++)
	    cin>>arr[j];
	    int k;
	    cin>>k;
	    int sum = 0,count=0;
	    unordered_map<int,int>m;
	    for(int j=0;j<n;j++){
	        sum += arr[j];
	        if(sum == k)
	        count++;
	        if(m.find(sum-k) != m.end())
	        count += m[sum-k];
	        m[sum]++;
	    }
	    cout<<count<<endl;
	}
	return 0;
}
